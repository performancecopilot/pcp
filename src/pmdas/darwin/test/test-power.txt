# Test Power/Battery statistics via IOPowerSources API
# These metrics provide battery and AC power information on macOS systems.
# On desktop Macs without batteries, battery_present=0 and metrics return safe defaults.

# Battery presence and charging state
desc power.battery.present
fetch power.battery.present

desc power.battery.charging
fetch power.battery.charging

# Battery charge level (0-100%)
desc power.battery.charge
fetch power.battery.charge

# Time remaining in minutes (-1 if not available/applicable)
desc power.battery.time_remaining
fetch power.battery.time_remaining

# Battery health percentage (0-100%)
desc power.battery.health
fetch power.battery.health

# Battery wear indicators
desc power.battery.cycle_count
fetch power.battery.cycle_count

# Battery temperature in °C×100 (e.g., 3542 = 35.42°C)
desc power.battery.temperature
fetch power.battery.temperature

# Battery electrical metrics
desc power.battery.voltage
fetch power.battery.voltage

desc power.battery.amperage
fetch power.battery.amperage

# Battery capacity in mAh
desc power.battery.capacity.design
fetch power.battery.capacity.design

desc power.battery.capacity.max
fetch power.battery.capacity.max

# AC power status
desc power.ac.connected
fetch power.ac.connected

# Current power source (string: "AC Power", "Battery Power", "Unknown")
desc power.source
fetch power.source
